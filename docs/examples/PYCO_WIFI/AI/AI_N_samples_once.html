17## WPC
18import pywpc
19
20## Parameters setting
21sampling_rate = 1000
22sample = 10
23delay = 2000
24
25## Start AI acquisition
26pywpc.AI_start(sampling_rate, sample)
27
28## Read data acquisition
29ai_sample = pywpc.AI_readNsample(sample, delay)
30print(ai_sample)
31
32## Stop AI acquisition
33pywpc.AI_stop()
34
35del ai_sample
