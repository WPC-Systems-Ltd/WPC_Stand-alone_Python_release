 4This example demonstrates how to plot FFT signal

 8Step2: Convert float to U8 list

19from ulab import numpy as np
20import pywpc
21
22## Parameters
23Fs = 256                ## sampling rate
24Ts = 1.0/Fs             ## sampling interval
25t = np.arange(0, 1, Ts) ## time vector
26frequency = 50          ## frequency of the signal
27
28## Sine signal
29signal = np.sin(2*np.pi*frequency*t)
30
31## Apply the FFT on the signal
32fourier_real, fourier_imaginary = np.fft.fft(signal)
33power_spectrum = fourier_real**2 + fourier_imaginary**2
34
35## Create list
36u8_list = []
37for i in power_spectrum:
38    u8_list.extend(pywpc.Sys_convertF32To4U8(i))
39
40## Connect to Wifi
41
42## Open TCP server
43pywpc.TCPServer_open(5566)
44
45## Send u8 list data
46pywpc.TCPServer_writeU8List(u8_list)
47
48## Close TCP server
49pywpc.TCPServer_close()
